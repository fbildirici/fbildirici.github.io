<div class="p-4 text-center">
\[
\phi_i(f,x)
=
\sum_{S\subseteq N\setminus\{i\}}
\frac{|S|!\,(|N|-|S|-1)!}{|N|!}\,
\Big(
\underbrace{\mathbb{E}\!\left[f(X)\mid X_{S\cup\{i\}}=x_{S\cup\{i\}}\right]}_{\text{with } i}
-
\underbrace{\mathbb{E}\!\left[f(X)\mid X_{S}=x_{S}\right]}_{\text{without } i}
\Big),
\quad i\in N,
\]
</div>
